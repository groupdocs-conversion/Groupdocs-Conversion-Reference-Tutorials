    class Program

        {

            string outputDirectory = "YOUR_OUTPUT_DIRECTORY";
            string outputFile = Path.Combine(outputDirectory, "docm-converted-to.xlsx");

            // Загрузите исходный файл DOCM с помощью GroupDocs.Conversion.
            using (var converter = new Converter(documentPath))
            {
                var options = new SpreadsheetConvertOptions(); // Параметры преобразования форматов электронных таблиц
                converter.Convert(outputFile, options); // Конвертировать и сохранить DOCM как XLSX
            }

            Console.WriteLine("Conversion completed successfully!");
        }
    }
}

## Руководство по внедрению
В этом разделе мы шаг за шагом разберем процесс конвертации.

### Конвертация DOCM в XLSX: пошаговое руководство
Эта функция позволяет вам легко конвертировать документы Microsoft Word с поддержкой макросов в таблицы Excel. Давайте углубимся в детали реализации.

#### Шаг 1: Определите пути к файлам и загрузите документ
Укажите путь к входному файлу для вашего документа DOCM и настройте выходной каталог для полученного файла XLSX.
```csharp
string documentPath = Path.Combine("YOUR_DOCUMENT_DIRECTORY", "sample.docm");
string outputDirectory = "YOUR_OUTPUT_DIRECTORY";
string outputFile = Path.Combine(outputDirectory, "docm-converted-to.xlsx");
```

#### Шаг 2: Инициализация GroupDocs.Conversion
Загрузите DOCM с помощью `Converter` класс предоставлен GroupDocs.
```csharp
using (var converter = new Converter(documentPath))
{
    // Логика преобразования будет располагаться здесь.
}
```

#### Шаг 3: Укажите параметры конвертации
Создать экземпляр `SpreadsheetConvertOptions`, который позволяет вам указывать параметры, адаптированные для форматов электронных таблиц, таких как XLSX.
```csharp
var options = new SpreadsheetConvertOptions();
```

#### Шаг 4: Выполнение преобразования
Используйте `Convert` метод для выполнения преобразования из DOCM в XLSX и сохранения выходного файла.
```csharp
converter.Convert(outputFile, options);
```

### Советы по устранению неполадок
- Убедитесь, что путь к входному файлу DOCM указан правильно.
- Убедитесь, что выходной каталог существует, или создайте его перед запуском преобразования.
- Проверьте, установлены ли необходимые разрешения на чтение входного документа и запись в выходное расположение.
