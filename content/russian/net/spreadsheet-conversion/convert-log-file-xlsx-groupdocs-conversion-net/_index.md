#### Базовая инициализация и настройка

Вот как можно инициализировать GroupDocs.Conversion в вашем приложении C#:

```csharp
using System;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

class Program
{
    static void Main(string[] args)
    {
        // Определите путь к документу и выходному каталогу.
        string documentPath = @"YOUR_DOCUMENT_DIRECTORY\example.log";
        
        using (Converter converter = new Converter(documentPath))
        {
            var convertOptions = new SpreadsheetConvertOptions();
            
            // Конвертировать в формат XLSX
            converter.Convert("output.xlsx", convertOptions);
        }
    }
}
```
