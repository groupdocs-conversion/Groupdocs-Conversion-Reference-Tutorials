### التهيئة والإعداد الأساسي
فيما يلي كيفية تهيئة GroupDocs.Conversion في مشروع C# الخاص بك:

```csharp
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

// تحديد مسارات المستندات
string documentDirectory = "YOUR_DOCUMENT_DIRECTORY";
string outputDirectory = "YOUR_OUTPUT_DIRECTORY/";

// تحديد مسارات ملفات الإدخال والإخراج
string inputFile = Path.Combine(documentDirectory, "sample.mobi");
string outputFile = Path.Combine(outputDirectory, "mobi-converted-to.pdf");

// تهيئة كائن المحول باستخدام ملف MOBI المصدر
using (var converter = new Converter(inputFile))
{
    // إعداد خيارات التحويل لتنسيق PDF
    var options = new PdfConvertOptions();
    
    // قم بإجراء التحويل من MOBI إلى PDF وحفظ النتيجة
    converter.Convert(outputFile, options);
}
```
