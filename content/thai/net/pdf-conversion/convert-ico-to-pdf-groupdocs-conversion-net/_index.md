## คู่มือการใช้งาน

### แปลงไฟล์ ICO เป็น PDF
#### ภาพรวม
คุณลักษณะนี้มุ่งเน้นไปที่การแปลงไฟล์ ICO เป็นรูปแบบ PDF โดยใช้ความสามารถอันแข็งแกร่งของ GroupDocs.Conversion สำหรับ .NET

**ขั้นตอนที่ 1: กำหนดเส้นทางต้นทางและปลายทางออก**
ก่อนอื่น ตรวจสอบให้แน่ใจว่าเส้นทางไฟล์ของคุณได้รับการกำหนดอย่างถูกต้อง:
```csharp
string outputFolder = "YOUR_OUTPUT_DIRECTORY";
string outputFile = Path.Combine(outputFolder, "ico-converted-to.pdf");

// สร้างอินสแตนซ์ตัวแปลงด้วยเส้นทางไฟล์ ICO
using (var converter = new Converter(inputFilePath))
{
    // กำหนดตัวเลือกสำหรับการแปลง PDF
    var options = new PdfConvertOptions();

    // แปลงและบันทึก ICO เป็นเอกสาร PDF
    converter.Convert(outputFile, options);
}
```

**ขั้นตอนที่ 2: การจัดการไดเรกทอรี**
การจัดการไดเร็กทอรีอย่างเหมาะสมเป็นสิ่งสำคัญเพื่อหลีกเลี่ยงปัญหาเส้นทางไฟล์:
```csharp
using System.IO;

// วิธีการเพื่อให้แน่ใจว่าไดเรกทอรีเอาท์พุตมีอยู่
string GetOutputDirectoryPath()
{
    string outputPath = Path.Combine("YOUR_OUTPUT_DIRECTORY");
    if (!Directory.Exists(outputPath))
    {
        Directory.CreateDirectory(outputPath);
    }
    
    return outputPath;
}
```
