### การเริ่มต้นและการตั้งค่าเบื้องต้น

เริ่มต้น GroupDocs.Conversion ในโครงการ C# ของคุณ:
```csharp
using System;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;
using GroupDocs.Conversion.Options.Load;

class Program
{
    static void Main()
    {
        // กำหนดไดเร็กทอรีเอาต์พุตและเส้นทางไฟล์
        string outputFolder = "YOUR_OUTPUT_DIRECTORY";
        string outputFile = Path.Combine(outputFolder, "converted.pdf");

        // เริ่มต้นตัวเลือกการโหลดเพื่อซ่อนความคิดเห็นในระหว่างการแปลง
        Func<LoadContext, LoadOptions> getLoadOptions = loadContext => new WordProcessingLoadOptions
        {
            CommentDisplayMode = WordProcessingCommentDisplay.Hidden // ตั้งค่าให้ความคิดเห็นถูกซ่อน
        };

        // แปลงเอกสารโดยใช้ GroupDocs.Conversion
        using (Converter converter = new Converter("YOUR_DOCUMENT_DIRECTORY/SAMPLE_DOCX_WITH_TRACKED_CHANGES", getLoadOptions))
        {
            PdfConvertOptions options = new PdfConvertOptions(); // เริ่มต้นตัวเลือกการแปลง PDF
            converter.Convert(outputFile, options); // ดำเนินการแปลงเป็น PDF
        }
    }
}
```

## คู่มือการใช้งาน

### แปลงเอกสารประมวลผลคำเป็น PDF พร้อมคำอธิบายประกอบที่ซ่อนอยู่

ปฏิบัติตามขั้นตอนเหล่านี้เพื่อแปลง `.docx` ไฟล์เป็น PDF พร้อมซ่อนความคิดเห็น

#### ขั้นตอนที่ 1: กำหนดตัวเลือกการโหลด
ชุด `CommentDisplayMode` ถึง `Hidden`-
```csharp
Func<LoadContext, LoadOptions> getLoadOptions = loadContext => new WordProcessingLoadOptions
{
    CommentDisplayMode = WordProcessingCommentDisplay.Hidden // ซ่อนความคิดเห็นระหว่างการแปลง
};
```

#### ขั้นตอนที่ 2: เริ่มต้นตัวแปลง
สร้าง `Converter` ตัวอย่าง:
```csharp
using (Converter converter = new Converter("YOUR_DOCUMENT_DIRECTORY/SAMPLE_DOCX_WITH_TRACKED_CHANGES", getLoadOptions))
{
    // ดำเนินการตั้งค่าการแปลง
}
```

#### ขั้นตอนที่ 3: ตั้งค่าตัวเลือกการแปลง PDF
การเริ่มต้น `PdfConvertOptions`-
```csharp
PdfConvertOptions options = new PdfConvertOptions();
```

#### ขั้นตอนที่ 4: ดำเนินการแปลง
ดำเนินการตามกระบวนการแปลง:
```csharp
converter.Convert(outputFile, options); // แปลงเป็น PDF พร้อมคำอธิบายที่ซ่อนอยู่
```
