### การเริ่มต้นและการตั้งค่าเบื้องต้น

ก่อนอื่นเรามาตั้งค่าโครงการกันก่อน นี่คือตัวอย่างโค้ดง่าย ๆ ที่ใช้เริ่มต้นไลบรารี GroupDocs.Conversion:

```csharp
using System;
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

class Program
{
    static void Main(string[] args)
    {
        // กำหนดไดเรกทอรีเอาท์พุต
        string outputFolder = Path.Combine("YOUR_OUTPUT_DIRECTORY");

        // ตรวจสอบให้แน่ใจว่ามีไดเร็กทอรีอยู่
        Directory.CreateDirectory(outputFolder);

        // เริ่มต้นตัวจัดการการแปลง
        using (var converter = new Converter("input.dwg"))
        {
            var options = new MarkupConvertOptions { Format = MarkupFileType.Tex };
            converter.Convert(Path.Combine(outputFolder, "output.tex"), options);
        }
    }
}
```

โค้ดนี้สาธิตสิ่งสำคัญในการโหลดไฟล์ DWG และแปลงเป็นรูปแบบ TEX

