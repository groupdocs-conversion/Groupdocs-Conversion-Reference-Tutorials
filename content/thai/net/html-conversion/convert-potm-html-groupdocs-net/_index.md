## คู่มือการใช้งาน
### แปลง POTM เป็น HTML
ปฏิบัติตามขั้นตอนเหล่านี้เพื่อแปลงไฟล์ .potm เป็น HTML:
#### ขั้นตอนที่ 1: กำหนดเส้นทางสำหรับไฟล์อินพุตและเอาต์พุต
ตั้งค่าไดเร็กทอรีสำหรับไฟล์ POTM อินพุตและไฟล์ HTML เอาท์พุต
```csharp
string documentDirectory = "YOUR_DOCUMENT_DIRECTORY"; // อัปเดตด้วยเส้นทางของคุณ
string outputDirectory = "YOUR_OUTPUT_DIRECTORY"; // อัปเดตด้วยเส้นทางของคุณ

class Constants
{
    public static string GetOutputDirectoryPath() => outputDirectory;
    
    public static string SAMPLE_POTM => Path.Combine(documentDirectory, "sample.potm");
}
```
#### ขั้นตอนที่ 2: ดำเนินการแปลง
แปลงไฟล์ POTM เป็น HTML โดยใช้ GroupDocs.Conversion
```csharp
using System.IO;
using GroupDocs.Conversion.Options.Convert;

string outputFile = Path.Combine(Constants.GetOutputDirectoryPath(), "potm-converted-to.html");

// เริ่มต้นตัวแปลงด้วยเส้นทางไฟล์อินพุต
using (var converter = new Converter(Constants.SAMPLE_POTM))
{
    // ระบุตัวเลือกการแปลงสำหรับรูปแบบ HTML
    var options = new WebConvertOptions();

    // ดำเนินการแปลงและบันทึกลงในไฟล์เอาท์พุตที่ระบุ
    converter.Convert(outputFile, options);
}
```
#### คำอธิบายแนวคิดหลัก
- **ตัวเลือกการแปลงเว็บ:** กำหนดค่าการตั้งค่าเฉพาะให้รูปแบบที่เป็นมิตรต่อเว็บ เช่น HTML
- **ตัวแปลง.แปลง():** เริ่มกระบวนการแปลงด้วยพารามิเตอร์ที่ให้ไว้
