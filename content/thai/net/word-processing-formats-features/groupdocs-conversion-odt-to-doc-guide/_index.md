## คู่มือการใช้งาน
ตอนนี้มาแบ่งการใช้งานออกเป็นขั้นตอนที่ชัดเจนกัน

### แปลง ODT เป็น DOC
ฟีเจอร์นี้สาธิตวิธีการแปลงไฟล์ Open Document Text (ODT) เป็นรูปแบบ Microsoft Word Document (DOC) โดยใช้ GroupDocs.Conversion สำหรับ .NET คุณสามารถทำได้ดังนี้:

#### ขั้นตอนที่ 1: กำหนดไดเรกทอรีและเส้นทางไฟล์
เริ่มต้นด้วยการระบุเส้นทางสำหรับไฟล์อินพุต ODT และไฟล์ DOC เอาท์พุตของคุณ

```csharp
string documentDirectory = @"YOUR_DOCUMENT_DIRECTORY"; // ไดเรกทอรีที่มีไฟล์ ODT ของคุณ
string outputDirectory = @"YOUR_OUTPUT_DIRECTORY"; // ไดเรกทอรีที่จะบันทึกไฟล์ DOC ที่แปลงแล้ว

// ตั้งค่าเส้นทางสำหรับไฟล์อินพุตและเอาท์พุต
string inputFile = Path.Combine(documentDirectory, "sample.odt");
string outputFile = Path.Combine(outputDirectory, "odt-converted-to.doc");
```

#### ขั้นตอนที่ 2: โหลดไฟล์ ODT ต้นฉบับ
ใช้ GroupDocs.Conversion เพื่อโหลดไฟล์ต้นฉบับของคุณ

```csharp
using (var converter = new GroupDocs.Conversion.Converter(inputFile))
{
    // ตรรกะการแปลงจะถูกเพิ่มที่นี่
}
```

#### ขั้นตอนที่ 3: ตั้งค่าตัวเลือกการแปลงสำหรับรูปแบบ DOC
ระบุว่าคุณต้องการแปลงไฟล์เป็นรูปแบบ DOC

```csharp
WordProcessingConvertOptions options = new WordProcessingConvertOptions { Format = GroupDocs.Conversion.FileTypes.WordProcessingFileType.Doc };
```

#### ขั้นตอนที่ 4: ดำเนินการแปลงและบันทึกไฟล์ผลลัพธ์
ดำเนินการขั้นตอนการแปลงและบันทึกผลลัพธ์ของคุณ

```csharp
converter.Convert(outputFile, options);
// ไฟล์ DOC ที่แปลงแล้วจะถูกบันทึกไว้ในไดเร็กทอรีเอาต์พุตที่ระบุแล้ว
```
