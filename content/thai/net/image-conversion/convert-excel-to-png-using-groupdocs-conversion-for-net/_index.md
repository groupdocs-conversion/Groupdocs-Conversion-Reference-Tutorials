## คู่มือการใช้งาน

ตอนนี้เรามาแปลงไฟล์ Excel เป็นภาพ PNG ทีละขั้นตอนกัน

### คุณสมบัติ: แปลง XLSX เป็น PNG
คุณลักษณะนี้สาธิตการแปลงเอกสาร XLSX ทั้งหมดเป็นไฟล์ PNG ชุดละหนึ่งไฟล์ต่อหนึ่งหน้าสเปรดชีต

#### ขั้นตอนที่ 1: กำหนดเส้นทางและโหลดไฟล์
ระบุตำแหน่งไฟล์ Excel ต้นทางและตำแหน่งที่คุณต้องการบันทึกรูปภาพเอาท์พุต:
```csharp
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

// กำหนดเส้นทางสำหรับไดเร็กทอรีอินพุตและเอาต์พุต (แทนที่ด้วยเส้นทางไดเร็กทอรีจริงของคุณ)
string documentDirectory = "YOUR_DOCUMENT_DIRECTORY";
string outputDirectory = "YOUR_OUTPUT_DIRECTORY";

// โหลดไฟล์ XLSX ต้นฉบับ
using (Converter converter = new Converter(Path.Combine(documentDirectory, "sample.xlsx")))
{
    // ตรรกะการแปลงจะไปที่นี่
}
```
#### ขั้นตอนที่ 2: ตั้งค่าตัวเลือกการแปลง
ระบุการแปลงแต่ละหน้า Excel ให้เป็นภาพ PNG:
```csharp
// ตั้งค่าตัวเลือกการแปลงสำหรับรูปแบบ PNG
ImageConvertOptions options = new ImageConvertOptions { Format = GroupDocs.Conversion.FileTypes.ImageFileType.Png };
```
#### ขั้นตอนที่ 3: กำหนดเส้นทางเอาต์พุตและแปลง
กำหนดชื่อไฟล์เอาท์พุต จากนั้นดำเนินการแปลง:
```csharp
// กำหนดเทมเพลตเส้นทางไฟล์เอาท์พุตสำหรับการแปลงแต่ละหน้า
string outputFileTemplate = Path.Combine(outputDirectory, "converted-page-{0}.png");

// สร้างฟังก์ชั่นเพื่อสร้างสตรีมสำหรับการบันทึกหน้าที่แปลงแล้ว
Func<SavePageContext, Stream> getPageStream = savePageContext =>
    new FileStream(string.Format(outputFileTemplate, savePageContext.Page), FileMode.Create);

// แปลงเอกสาร XLSX เป็นภาพ PNG โดยใช้ตัวเลือกและเครื่องกำเนิดสตรีมเอาท์พุตที่ระบุ
converter.Convert(getPageStream, options);
```
### เคล็ดลับการแก้ไขปัญหา
- **ข้อผิดพลาดไม่พบไฟล์**: ตรวจสอบให้แน่ใจว่าเส้นทางไฟล์ถูกต้องและสามารถเข้าถึงได้
- **ปัญหาการอนุญาต**: ตรวจสอบสิทธิ์การอ่าน/เขียนสำหรับไดเร็กทอรี
