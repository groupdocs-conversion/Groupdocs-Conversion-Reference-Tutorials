## คู่มือการใช้งาน

### ฟีเจอร์แปลง GIF เคลื่อนไหวเป็น PPTX

ฟีเจอร์นี้ช่วยให้สามารถแปลงไฟล์ GIF เป็นรูปแบบการนำเสนอ PowerPoint ได้ มาดูขั้นตอนที่เกี่ยวข้องกัน

#### ขั้นตอนที่ 1: กำหนดเส้นทางไฟล์และตั้งค่าตัวแปลง

ก่อนอื่น ให้กำหนดไดเร็กทอรีอินพุตและเอาต์พุตของคุณ:

```csharp
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

string outputFolder = "YOUR_OUTPUT_DIRECTORY"; // กำหนดเส้นทางไดเรกทอรีเอาท์พุต
string inputFile = Path.Combine("YOUR_DOCUMENT_DIRECTORY", "sample.gif"); // เส้นทางเข้าไฟล์ GIF
string outputFile = Path.Combine(outputFolder, "gif-converted-to.pptx"); // กำหนดชื่อไฟล์ PPTX เอาท์พุต

// โหลดไฟล์ GIF ต้นฉบับ
using (var converter = new Converter(inputFile))
{
    var options = new PresentationConvertOptions(); // ตั้งค่าตัวเลือกการแปลงสำหรับรูปแบบ PowerPoint
}
```

#### ขั้นตอนที่ 2: ดำเนินการแปลงและบันทึกผลลัพธ์

ภายในบล็อกการใช้งาน ดำเนินการแปลง:

```csharp
converter.Convert(outputFile, options); // แปลงและบันทึกไฟล์ PPTX
Console.WriteLine("Conversion Successful!");
```

1. **การนำเสนอการตลาด**:แปลง GIF ส่งเสริมการขายเป็นภาพสไลด์โชว์สำหรับการนำเสนอต่อลูกค้า
2. **เนื้อหาการศึกษา**:แปลงสื่อการสอนจาก GIF สู่การบรรยาย PowerPoint แบบโต้ตอบ
3. **รายงานทางธุรกิจ**:ใช้การแสดงภาพข้อมูลแบบเคลื่อนไหวในรายงานขององค์กรโดยแปลงเป็น PPTX
