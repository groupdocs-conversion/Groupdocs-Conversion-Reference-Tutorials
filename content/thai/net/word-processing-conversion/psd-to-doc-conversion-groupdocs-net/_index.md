### คุณสมบัติ: โหลดและแปลง PSD เป็น DOC
#### ภาพรวม
ฟีเจอร์นี้ช่วยให้สามารถโหลดไฟล์ Photoshop Design (PSD) และแปลงเป็น Microsoft Word Document (DOC) ได้ ซึ่งมีประโยชน์สำหรับนักออกแบบที่ต้องการแบ่งปันผลงานของตนในรูปแบบข้อความที่แก้ไขได้

#### ขั้นตอนที่ 1: โหลดไฟล์ PSD ต้นฉบับ

```csharp
using System.IO;
using GroupDocs.Conversion.Options.Convert;

// กำหนดไดเรกทอรีเอกสารและโฟลเดอร์เอาท์พุตของคุณ
string documentDirectory = @"YOUR_DOCUMENT_DIRECTORY";
string psdFilePath = Path.Combine(documentDirectory, "your-source-file.psd");
string outputFolder = @"YOUR_OUTPUT_DIRECTORY";
string outputFile = Path.Combine(outputFolder, "psd-converted-to.doc");

// โหลดไฟล์ PSD โดยใช้ GroupDocs.Conversion
using (var converter = new Converter(psdFilePath))
{
    // กระบวนการแปลงจะถูกจัดการในขั้นตอนถัดไป
}
```

#### ขั้นตอนที่ 2: ตั้งค่าตัวเลือกการแปลง

```csharp
// กำหนดค่าตัวเลือกการแปลงสำหรับรูปแบบการประมวลผลคำ
var options = new WordProcessingConvertOptions { Format = WordProcessingFileType.Doc };
```

#### ขั้นตอนที่ 3: แปลงและบันทึกไฟล์

```csharp
// ดำเนินการแปลงและบันทึกผลลัพธ์เป็นไฟล์ DOC
converter.Convert(outputFile, options);
```

สไนปเป็ตนี้จะเริ่มต้นกระบวนการแปลงโดยการโหลดไฟล์ PSD ตั้งค่าตัวเลือกการแปลงที่เหมาะสมสำหรับรูปแบบ DOC และดำเนินการแปลง
