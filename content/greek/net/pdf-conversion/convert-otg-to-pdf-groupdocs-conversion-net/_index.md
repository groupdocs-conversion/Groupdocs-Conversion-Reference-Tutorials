```csharp
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

// Καθορίστε τις διαδρομές καταλόγου
cstring outputFolder = Path.Combine(@"YOUR_OUTPUT_DIRECTORY");
cstring outputFile = Path.Combine(outputFolder, "otg-converted-to.pdf");

// Φορτώστε το αρχείο OTG (βεβαιωθείτε ότι αυτή η διαδρομή είναι σωστή)
cstring otgFilePath = @"YOUR_DOCUMENT_DIRECTORY\sample.otg";

// Αρχικοποιήστε τον μετατροπέα με το αρχείο OTG
using (var converter = new Converter(otgFilePath))
{
    var options = new PdfConvertOptions(); // Ρύθμιση παραμέτρων επιλογών μετατροπής PDF
    converter.Convert(outputFile, options); // Μετατροπή και αποθήκευση του PDF
}
```
