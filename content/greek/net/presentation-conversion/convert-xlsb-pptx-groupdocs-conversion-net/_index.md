```csharp
using System;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

namespace YourNamespace
{
    class Program
    {
        static void Main(string[] args)
        {
            // Αρχικοποίηση ενός νέου αντικειμένου Converter με τη διαδρομή του αρχείου προέλευσης
            using (var converter = new Converter("path/to/your/source/file.xlsb"))
            {
                var options = new PresentationConvertOptions();
                
                // Μετατρέψτε και αποθηκεύστε την έξοδο σε μορφή PPTX
                converter.Convert("output/path/converted-file.pptx", options);
            }
        }
    }
}
```

