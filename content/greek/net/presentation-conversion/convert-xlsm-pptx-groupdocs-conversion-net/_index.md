### Βασική Αρχικοποίηση και Ρύθμιση

Δείτε πώς μπορείτε να αρχικοποιήσετε και να ρυθμίσετε το GroupDocs.Conversion σε ένα έργο C#:

```csharp
using System;
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

// Αρχικοποιήστε το αντικείμενο Converter με τη διαδρομή αρχείου XLSM
string inputFile = "path/to/your/sample.xlsm";
string outputFolder = "path/to/output/directory";

// Δημιουργήστε μια παρουσία της κλάσης Converter
using (var converter = new Converter(inputFile))
{
    // Ορίστε επιλογές μετατροπής για τη μορφή PPTX
    var options = new PresentationConvertOptions();
    
    // Καθορίστε τη διαδρομή του αρχείου εξόδου
    string outputFile = Path.Combine(outputFolder, "xlsm-converted-to.pptx");
    
    // Εκτελέστε τη μετατροπή και αποθηκεύστε το αρχείο PPTX που προκύπτει
    converter.Convert(outputFile, options);
}
```
