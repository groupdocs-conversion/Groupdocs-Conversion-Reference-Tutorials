```csharp

using GroupDocs.Conversion;

class Program
{
    static void Main()
    {
        string documentPath = "sample.xlsx";
        string outputFile = "output.pdf";

        using (var converter = new Converter(documentPath))
        {

            converter.Convert(outputFile, options);
        }

    }
}
```
