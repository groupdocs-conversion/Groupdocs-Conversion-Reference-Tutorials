#### Βασική Αρχικοποίηση και Ρύθμιση
Για να αρχικοποιήσετε το GroupDocs.Conversion στην εφαρμογή C# που διαθέτετε, χρησιμοποιήστε την ακόλουθη ρύθμιση:
```csharp
using System;
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

namespace DocmToXlsxConversion
{
    class Program
    {
        static void Main(string[] args)
        {
            string documentPath = Path.Combine("YOUR_DOCUMENT_DIRECTORY", "sample.docm");
            string outputDirectory = "YOUR_OUTPUT_DIRECTORY";
            string outputFile = Path.Combine(outputDirectory, "docm-converted-to.xlsx");

            // Φορτώστε το αρχείο DOCM πηγής χρησιμοποιώντας το GroupDocs.Conversion.
            using (var converter = new Converter(documentPath))
            {
                var options = new SpreadsheetConvertOptions(); // Επιλογές μετατροπής για μορφές υπολογιστικών φύλλων
                converter.Convert(outputFile, options); // Μετατροπή και αποθήκευση του DOCM ως XLSX
            }

            Console.WriteLine("Conversion completed successfully!");
        }
    }
}
```
