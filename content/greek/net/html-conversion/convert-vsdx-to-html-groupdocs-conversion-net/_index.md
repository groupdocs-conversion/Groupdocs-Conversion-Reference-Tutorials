### Βασική Αρχικοποίηση και Ρύθμιση

Δείτε πώς μπορείτε να αρχικοποιήσετε το GroupDocs.Conversion στο έργο σας C#:

```csharp
using System;
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

// Ορισμός καταλόγου εξόδου και διαδρομής αρχείου
string outputFolder = Path.Combine("YOUR_OUTPUT_DIRECTORY");
string outputFile = Path.Combine(outputFolder, "vsdx-converted-to.html");

// Φόρτωση του αρχείου πηγής VSDX
using (var converter = new GroupDocs.Conversion.Converter(@"YOUR_DOCUMENT_DIRECTORY\\sample.vsdx"))
{
    // Αρχικοποίηση επιλογών μετατροπής για HTML
    var options = new WebConvertOptions();
    
    // Μετατροπή και αποθήκευση του VSDX σε μορφή HTML
    converter.Convert(outputFile, options);
}
```
