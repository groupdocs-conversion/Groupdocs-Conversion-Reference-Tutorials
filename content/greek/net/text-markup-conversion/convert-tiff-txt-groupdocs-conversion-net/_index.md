#### Αρχικοποίηση και Ρύθμιση
Για να αρχικοποιήσετε το GroupDocs.Conversion στο έργο σας, προσθέστε το ακόλουθο απόσπασμα κώδικα:
```csharp
using System;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

class Program
{
    static void Main()
    {
        // Ρύθμιση παραμέτρων των διαδρομών πηγής και εξόδου
        string sourceTiffPath = Path.Combine("YOUR_DOCUMENT_DIRECTORY", "sample.tiff");
        string outputPath = Path.Combine("YOUR_OUTPUT_DIRECTORY", "output.txt");

        // Αρχικοποίηση μετατροπέα
        using (Converter converter = new Converter(sourceTiffPath))
        {
            // Ρύθμιση επιλογών μετατροπής για μορφή TXT
            var convertOptions = new TextConvertOptions();
            
            // Μετατρέψτε και αποθηκεύστε το αρχείο εξόδου
            converter.Convert(outputPath, convertOptions);
        }
    }
}
```
