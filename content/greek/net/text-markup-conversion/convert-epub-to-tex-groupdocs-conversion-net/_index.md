### Βασική Αρχικοποίηση

Ξεκινήστε ρυθμίζοντας το βασικό περιβάλλον για τη φόρτωση και τη μετατροπή των εγγράφων σας. Ακολουθεί μια αρχική ρύθμιση σε C#:

```csharp
using System;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

class Program
{
    static void Main(string[] args)
    {
        string DocumentPath = "YOUR_DOCUMENT_DIRECTORY/sample.epub";
        string OutputDirectory = "YOUR_OUTPUT_DIRECTORY";

        using (var converter = new Converter(DocumentPath))
        {
            var options = new PageDescriptionLanguageConvertOptions { Format = PageDescriptionLanguageFileType.Tex };
            string outputFile = System.IO.Path.Combine(OutputDirectory, "epub-converted-to.tex");
            converter.Convert(outputFile, options);
        }
    }
}
```
