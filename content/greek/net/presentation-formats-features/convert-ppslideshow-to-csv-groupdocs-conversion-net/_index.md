### Βασική Αρχικοποίηση και Ρύθμιση
Για να ξεκινήσετε τη μετατροπή αρχείων χρησιμοποιώντας C#, αρχικοποιήστε το περιβάλλον μετατροπής σας όπως φαίνεται παρακάτω:

```csharp
using System;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

namespace PpsToCsvConversion {
    internal static class ConvertPpsToCsvFeature {
        public static void Run() {
            // Ορίστε τη διαδρομή του καταλόγου εξόδου.
            string outputFolder = "YOUR_OUTPUT_DIRECTORY"; 
            string outputFile = System.IO.Path.Combine(outputFolder, "pps-converted-to.csv");

            using (var converter = new Converter("YOUR_DOCUMENT_DIRECTORY\\\\sample.pps")) {
                var options = new SpreadsheetConvertOptions { Format = SpreadsheetFileType.Csv };
                converter.Convert(outputFile, options);
            }
        }
    }
}
```
