```csharp
using System;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

class Program
{
    static void Main()
    {
        // Ορίστε τη διαδρομή του καταλόγου εξόδου και τη θέση του αρχείου CDR
        string outputFolder = "YOUR_OUTPUT_DIRECTORY";
        string sampleCdrPath = "YOUR_DOCUMENT_DIRECTORY\sample.cdr"; // Αντικαταστήστε με την πραγματική σας διαδρομή

        string outputFile = Path.Combine(outputFolder, "cdr-converted-to.pptx");

        // Αρχικοποιήστε το αντικείμενο μετατροπέα με το αρχείο CDR πηγής
        using (var converter = new Converter(sampleCdrPath))
        {
            var options = new PresentationConvertOptions();
            
            // Μετατρέψτε και αποθηκεύστε το αρχείο PPTX εξόδου
            converter.Convert(outputFile, options);
        }
    }
}
```
