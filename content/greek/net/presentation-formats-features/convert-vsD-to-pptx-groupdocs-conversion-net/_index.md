### Βασική Αρχικοποίηση και Ρύθμιση
Δείτε πώς μπορείτε να αρχικοποιήσετε το GroupDocs.Conversion στο έργο σας C#:
```csharp
using System;
using GroupDocs.Conversion;

class Program
{
    static void Main(string[] args)
    {
        string inputFilePath = @"path\to\your\sample.vsd"; // Καθορίστε τη διαδρομή προς το αρχείο VSD σας
        string outputFolder = @"path\to\output\directory"; // Κατάλογος εξόδου για τα αρχεία που έχουν μετατραπεί

        using (var converter = new Converter(inputFilePath))
        {
            var options = new PresentationConvertOptions(); // Αρχικοποίηση επιλογών μετατροπής για το PowerPoint

            // Μετατρέψτε και αποθηκεύστε το αρχείο σε μορφή PPTX
            string outputFile = System.IO.Path.Combine(outputFolder, "vsd-converted-to.pptx");
            converter.Convert(outputFile, options);
        }
    }
}
```
