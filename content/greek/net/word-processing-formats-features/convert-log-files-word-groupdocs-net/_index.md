### Βασική Αρχικοποίηση και Ρύθμιση

Δείτε πώς μπορείτε να αρχικοποιήσετε τη βιβλιοθήκη GroupDocs.Conversion στο έργο σας C#:

```csharp
using System;
using System.IO;
using GroupDocs.Conversion;
using GroupDocs.Conversion.Options.Convert;

namespace LogToDocConverter
{
    class Program
    {
        static void Main(string[] args)
        {
            // Ορισμός καταλόγων εισόδου και εξόδου
            string documentDirectory = "YOUR_DOCUMENT_DIRECTORY";
            string logFilePath = Path.Combine(documentDirectory, "example.log");
            string outputDirectory = "YOUR_OUTPUT_DIRECTORY";

            // Αρχικοποίηση μετατροπέα
            using (var converter = new Converter(logFilePath))
            {
                var convertOptions = new WordProcessingConvertOptions();
                
                // Μετατροπή και αποθήκευση του εγγράφου
                converter.Convert(Path.Combine(outputDirectory, "converted.doc"), convertOptions);
            }
        }
    }
}
```
